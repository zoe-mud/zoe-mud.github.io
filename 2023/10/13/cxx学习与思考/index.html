调用静态成员函数或变量，使用类名::函数名，非静态成员函数或变量，使用实例.函数名

class Entity{
public:
    //静态成员变量
    static int s;
     
    //非静态成员变量
    int n;
    
    //构造函数
    MyClass(int n)
    {
        this->n = n;
    }
    
    //静态成员函数
    static void staticMethod()
    {
        
    }
    
    //非静态成员函数
    void nonStaticMethod()
    {
        
    }
};

//初始化静态成员变量
int Entity::s = 10;

int main()
{
    cout<<"静态成员变量"<<Entity::s;
    
    cout<<"非静态成员变量"<<Entity::n;
    
    //创建对象
    Entity e(20);
    
    //调用静态函数
    Entity::staticMethod();
    
    //调用非静态函数
    e.nonStaticMethod();
    
    return 0;
        
}

->：通过指针访问对象的成员，指针->成员名

.：通过对象本身访问成员，对象.成员名

c++中this的作用：指向成员函数所作用的对象。所以非静态成员函数中可以直接使用this来代表指向该函数作用的对象的指针。

注：实际上每个成员函数的第一个参数默认都有个指向对象的指针this，this指针为空时，即表示所指向对象为空，相当于成员函数的第一个参数是NULL，那么只要成员函数没有使用到成员变量，也是可以正常执行。

静态成员函数不能使用this指针，因为静态成员函数相当于是共享的变量，不属于某个对象的变量。

this的本质是指针常量。

重载<<运算符

#include<iostream>
using namespace std;
class Entity{
private:
    int x,y;
public:
    Entity(){}
    Entity(int a,int b):x(a),y(b){}
    ~Entity(){}
    friend ostream& operator <<(ostream& os,const Entity& e);
    friend istream& operator>>(istream &is,Entity& e)；
};
ostream& operator <<(ostream& os,const Entity& e){
    os<<e.x<<e.y;
    return os;
}
istream& operator>>(istream &is,Entity& e){
    is>>e.x>>e.y
    return is：
}
int main(){
 Entity e(3,4);
 cout<<"e："<<e<<endl;
 Entity e1;
 cin>>e1;
 cout<<"e1："<<e1<<endl;    
}

讲解：

第一个参数 os 是一个引用类型的 ostream 对象，它代表了输出流。第二个参数 obj 是我们想要输出的对象，它通常是一个常量引用类型。

由于在输出时需要修改 os 对象本身，因此需要使用引用类型。而返回值类型为 ostream&，表示返回了一个 ostream 对象的引用，以便于进行链式操作。

因为要访问类的私有的成员变量，所以使用friend关键字，让重载函数称为类的友元函数。

模板

模板形参表示的是一个未知的类型,比如：int,Entity

函数模板

模板声明或定义只能在全局，命名空间或类范围

函数模板是不会进行隐式数据类型转换

当函数模板和普通函数都符合调用时，优先选择普通重载函数

#include<iostream>
using namespace std;

template <class T,class T1> 
void ADD( T& a,  T1& b) {//使用引用可以不用创建副本
    a--;
    b++;
}
int main() {
    int x = 10;
    double y = 11.1;
    //ADD(x, y);//自动数据推导
    ADD<int, double>(x, y);
    cout << "x:" << x << endl;
    cout << "y:" << y << endl;
}

类模板

  • 创建对象方式

    默认构造函数

    MyClass obj;  // 使用默认构造函数创建对象

    参数化构造函数

    MyClass obj(10, 3.14);  // 使用参数化构造函数创建对象，并传入参数

    拷贝构造函数

    MyClass obj1(10, 3.14);
MyClass obj2(obj1);  // 使用拷贝构造函数创建对象

    赋值运算符

    MyClass obj1(10, 3.14);
MyClass obj2 = obj1;  // 使用赋值运算符创建对象

    结合new运算符动态创建对象，并返回指向该对象的指针

    MyClass* ptr = new MyClass(10, 3.14);  // 动态创建对象，并返回指针

delete ptr;  // 手动释放动态创建的对象所占用的内存

  • 类模板示例

  • #include<iostream>
using namespace std;
class Entity {
public:
    T a, b;
    T swap1(T& a, T& b){
    T c;
    c = a;
    a = b;
    b = c;
    }
};
int main() {
    Entity<char>* e0 = new Entity<char>('a','b');//new运算符动态创建对象，并返回指向该对象的指针
    Entity<double> e1(3.4,2.5);
    Entity<double> e2(e1);
    e0->swap1();
    e2.swap1();
    cout << e0->a << endl;

    cout << e2.a << endl;
    delete e0;
}
